## 1. Exemple clair et compact

```python
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QAction

class Fenetre(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Menu minimal")
        self.setGeometry(100, 100, 400, 300)

        # Menu principal
        barre = self.menuBar()
        menu_fichier = barre.addMenu("Fichier")

        # Actions
        action_nouveau = QAction("Nouveau", self)
        action_nouveau.triggered.connect(self.nouveau)
        menu_fichier.addAction(action_nouveau)

        menu_fichier.addSeparator()

        action_quitter = QAction("Quitter", self)
        action_quitter.triggered.connect(self.close)
        menu_fichier.addAction(action_quitter)

    def nouveau(self):
        print("Nouveau fichier...")

app = QApplication(sys.argv)
fenetre = Fenetre()
fenetre.show()
app.exec()
```

## 2.  À retenir :

menuBar() → crée la barre
.addMenu() → ajoute un menu
QAction() → crée une action
.addAction() → ajoute l’action au menu
.triggered.connect() → lie à une méthode
